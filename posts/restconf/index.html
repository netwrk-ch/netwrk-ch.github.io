Base URL for all the following examples https://{HOST}:{PORT}/restconf/data/

Get a Portchannel Subinterface Cisco-IOS-XE-native:native/interface/Port-channel-subinterface/Port-channel=1.600

Delete a Portchannel Subinterface Cisco-IOS-XE-native:native/interface/Port-channel-subinterface/Port-channel=1.600

Operational information about all the vrfs Cisco-IOS-XE-vrf-oper:vrf-oper-data

Get the information for a specific vrf Cisco-IOS-XE-native:native/vrf/definition=TEST

Python examples

These are really simple examples that can hopefully help you to get on speed (and help me remember how this stuff works) :)

Create a new VRF

This examples creates a new vrf on the defined switch. It only works when the vrf doesn’t already exists.

import requests

requests.packages.urllib3.disable_warnings()

HOST = '1.1.1.1'
PORT = 443
USER = 'user'
PASS = 'password'

def main():
    url = f'https://{HOST}:{PORT}/restconf/data/Cisco-IOS-XE-native:native/vrf'

    headers = {
        'Content-Type': 'application/yang-data+json',
        'Accept': 'application/yang-data+json'
    }

    data = {
        "Cisco-IOS-XE-native:definition": {
            "name": "RESTCONF"
        }
    }

    response = requests.post(
        url,
        auth=(USER, PASS),
        headers=headers,
        verify=False,
        json=data
    )

    print(response)

if __name__ == '__main__':
    main()

import requests

requests.packages.urllib3.disable_warnings()

HOST = '1.1.1.1'
PORT = 443
USER = 'user'
PASS = 'password'

def main():
    url = f'https://{HOST}:{PORT}/restconf/data/Cisco-IOS-XE-native:native'

    headers = {
        'Content-Type': 'application/yang-data+json',
        'Accept': 'application/yang-data+json'
    }

    data = {
        "native":{
            "vrf": {
                "definition": {
                    "name": "TEST",
                    "address-family": {
                        "ipv4": {
                            "route-target": {
                                "import-route-target": {
                                    "without-stitching": [
                                        {
                                            "asn-ip": "5:5"
                                        }
                                    ]
                                }
                            }
                        }
                    }
                }
            }
        }
    }

    response = requests.patch(
        url,
        auth=(USER, PASS),
        headers=headers,
        verify=False,
        json=data
    )

    print(response.text)

if __name__ == '__main__':
    main()
